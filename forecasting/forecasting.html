# save_delhi_geojson.py
import geopandas as gpd

# Read the shapefile
shapefile_path = "data/shapefile/Districts.shp"
delhi = gpd.read_file(shapefile_path)

# Simplify the geometry to reduce points (speeds up web rendering)
delhi_simple = delhi.copy()
delhi_simple['geometry'] = delhi_simple['geometry'].simplify(tolerance=0.001, preserve_topology=True)

# Save as GeoJSON
delhi_simple.to_file("data/shapefile/delhi_simple_001.geojson", driver="GeoJSON")

print("GeoJSON with tolerance = 0.001")
